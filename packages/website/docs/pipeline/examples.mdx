  本页中的拓展功能依赖于 fusion 组件库，但引入 fusion 会影响全局样式与暗色主题，所以当前文档页的样式现在比较糟糕。

</b>

### 树形表格

```jsx live
function 树形表格() {
  const { dataSource4, columns4 } = assets.biz
  const pipeline = useTablePipeline({ components: fusion })
    .input({ dataSource: dataSource4, columns: columns4 })
    .primaryKey('id')
    .use(features.treeMode({ defaultOpenKeys: ['4', '4-2'] }))

  return (
    <>
      <assets.WithCSS library="fusion" />
      <BaseTable {...pipeline.getProps()} />
    </>
  )
}
```

### 行多选

```jsx live
function 行多选() {
  const { dataSource4, columns4 } = assets.biz

  const pipeline = useTablePipeline({ components: fusion })
    .input({ dataSource: dataSource4, columns: columns4 })
    .primaryKey('id')
    .use(
      features.multiSelect({
        highlightRowWhenSelected: true,
        defaultValue: { keys: ['1', '2'], lastKey: '1' },
      }),
    )

  return (
    <>
      <p>按下 shift 来选择多行</p>
      <assets.WithCSS library="fusion" />
      <BaseTable {...pipeline.getProps()} />
    </>
  )
}
```

### 行单选

```jsx live
function 行单选() {
  const { dataSource1, columns1 } = assets.biz

  const pipeline = useTablePipeline({ components: fusion })
    .input({ dataSource: dataSource1, columns: columns1 })
    .primaryKey('id')
    .use(
      features.singleSelect({
        defaultValue: '1',
        highlightRowWhenSelected: true,
        checkboxPlacement: 'start',
        clickArea: 'row',
      }),
    )

  return (
    <>
      <assets.WithCSS library="fusion" />
      <BaseTable {...pipeline.getProps()} />
    </>
  )
}
```

### 行展开

```jsx live
function 行展开() {
  const { dataSource5, columns5 } = assets.biz

  const pipeline = useTablePipeline({ components: fusion })
    .input({
      // dataSource 结构：
      //  [ { id: '1', title: '阿里巴巴网络技术有限公司', children: [...] },
      //    { id: '2', title: '蚂蚁金服有限公司', children: [...] },
      //    { id: '3', title: '其他', children: [...] }, ]
      dataSource: dataSource5,
      columns: columns5,
    })
    .primaryKey('id')
    // 受控用法：rowExpand({ openKeys, onChangeOpenKeys })
    .use(features.rowExpand({ defaultOpenKeys: ['1', '2'] }))

  return (
    <>
      <assets.WithCSS library="fusion" />
      <BaseTable {...pipeline.getProps()} />
    </>
  )
}
```
